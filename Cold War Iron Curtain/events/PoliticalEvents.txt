	desc = political.1.d

		NOT = { has_government = communism }

		name = political.1.a

				NOT = { tag = CHI }

	}
}

country_event = {
	id = political.10
	title = political.10.t
	desc = political.10.d
	picture = GFX_report_event_fascist_gathering

	is_triggered_only = yes

	option = {
		name = political.10.a
		if = {
			limit = {
				has_government = democratic
			}
			if = {
				limit = { fascism > 0.4 }
				start_civil_war = {
					ruling_party = fascism
					ideology = democratic
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { fascism > 0.4 } }
				start_civil_war = {
					ruling_party = fascism
					ideology = democratic
					size = 0.7
				}
			}
		}
		if = {
			limit = {
				has_government = communism
			}
			if = {
				limit = { fascism > 0.4 }
				start_civil_war = {
					ruling_party = fascism
					ideology = communism
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { fascism > 0.4 } }
				start_civil_war = {
					ruling_party = fascism
					ideology = communism
					size = 0.7
				}
			}
		}
		if = {
			limit = {
				has_government = neutrality
