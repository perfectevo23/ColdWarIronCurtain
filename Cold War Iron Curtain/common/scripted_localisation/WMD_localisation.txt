#STATUS BUTTONS


defined_text = { 

	name = target_status
	
	#not selected
	
	text = {
		trigger = {
			OR = {
				check_variable = { target_selected = 0}
				check_variable = { ROOT.nations_selected = 0 }
			}
		}
		localization_key = GFX_wmd_status_target
	}
	
	#selected
	
	text = {
		trigger = {
			check_variable = { target_selected = 1}
			check_variable = { ROOT.nations_selected > 0 }
		}
		localization_key = GFX_wmd_status_target_yes
	}
	
}

defined_text = { 

	name = missile_status
	
	#not selected
	
	text = {
		trigger = {
			OR = {
				check_variable = { missile_selected = 0}
				check_variable = { ROOT.nations_selected = 0 }
			}
		}
		localization_key = GFX_wmd_status_missile
	}
	
	#selected
	
	text = {
		trigger = {
			AND = {
				check_variable = { missile_selected = 1}
				check_variable = { ROOT.nations_selected > 0 }
			}
		}
		localization_key = GFX_wmd_status_missile_yes
	}
	
}
